if (NOT ${JK_GA_INC} MATCHES "NONE")
   message( "Using Global Arrays with includes path: ${JK_GA_INC}" )
   message( "Using Global Arrays with libraries path: ${JK_GA_LIB}" )
else()
   include(ExternalProject)
   get_filename_component(GA_BUILD_DIR ${CMAKE_BINARY_DIR} ABSOLUTE)
   set(GA_SRC ${CMAKE_BINARY_DIR}/ga-5-3b_Src)
   ExternalProject_Add(GlobalArrays
      #DOWNLOAD_COMMAND ${CMAKE_COMMAND} -E tar xzf 
      URL   ${CMAKE_CURRENT_SOURCE_DIR}/ga-5-3b.tgz 
      CONFIGURE_COMMAND <SOURCE_DIR>/configure 
         F77=${CMAKE_FORTRAN_COMP} 
         CC=${CMAKE_C_COMPILER} 
         CXX=${CMAKE_CXX_COMPILER} 
         --prefix=${GA_BUILD_DIR}
      BUILD_COMMAND make     
      INSTALL_COMMAND make install
  )
  set(JK_GA_INC ${GA_BUILD_DIR}/include)
  set(JK_GA_LIB ${GA_BUILD_DIR}/lib)
  message("GA installed in: ${JK_GA_LIB}")
  add_library(ga STATIC IMPORTED)
  add_library(armci STATIC IMPORTED)
  set_target_properties(ga PROPERTIES IMPORTED_LOCATION
       (${CMAKE_BINARY_DIR}/lib))
   set_target_properties(armci PROPERTIES IMPORTED_LOCATION
       (${CMAKE_BINARY_DIR}/lib))
endif()
